SELECT count(*)
FROM "profile_latest_application" AS p0
         LEFT OUTER JOIN "profiles" AS p1 ON p1."user_id" = p0."submitted_by"
         LEFT OUTER JOIN "channel" AS c2 ON c2."id" = p1."channel_id"
WHERE (p1."deleted" = False)
  AND (c2."default" = True)
  AND (p0."status" = 'submitted');
